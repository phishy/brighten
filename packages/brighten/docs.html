<div id="editor"></div>

<script type="module">
  import { EditorUI } from 'brighten';

  const editor = new EditorUI({
    container: '#editor',
    image: './photo.jpg',
    theme: 'dark',
    onExport: (blob) => {
      // Download the edited image
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url;
      a.download = 'edited.png';
      a.click();
    }
  });
</script>
import { useEffect, useRef } from 'react';
import { EditorUI } from 'brighten';

export default function PhotoEditor({ src }: { src: string }) {
  const containerRef = useRef<HTMLDivElement>(null);
  const editorRef = useRef<EditorUI | null>(null);

  useEffect(() => {
    if (!containerRef.current) return;
    
    editorRef.current = new EditorUI({
      container: containerRef.current,
      image: src,
      theme: 'dark',
      onExport: (blob) => console.log('Exported:', blob)
    });

    return () => editorRef.current?.destroy();
  }, [src]);

  return <div ref={containerRef} style={{ height: '600px' }} />;
}
<script setup lang="ts">
import { ref, onMounted, onUnmounted } from 'vue';
import { EditorUI } from 'brighten';

const props = defineProps<{ src: string }>();
const container = ref<HTMLElement>();
let editor: EditorUI | null = null;

onMounted(() => {
  if (!container.value) return;
  editor = new EditorUI({
    container: container.value,
    image: props.src,
    theme: 'dark',
    onExport: (blob) => console.log('Exported:', blob)
  });
});

onUnmounted(() => editor?.destroy());
</script>

<template>
  <div ref="container" style="height: 600px" />
</template>
'use client';
import { useEffect, useRef } from 'react';
import dynamic from 'next/dynamic';

function PhotoEditor({ src }: { src: string }) {
  const containerRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (!containerRef.current) return;
    
    // Dynamic import for client-side only
    import('brighten').then(({ EditorUI }) => {
      new EditorUI({
        container: containerRef.current!,
        image: src,
        theme: 'dark',
        onExport: (blob) => console.log('Exported:', blob)
      });
    });
  }, [src]);

  return <div ref={containerRef} style={{ height: '600px' }} />;
}

export default dynamic(() => Promise.resolve(PhotoEditor), { 
  ssr: false 
});
